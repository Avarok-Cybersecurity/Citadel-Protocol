1//! Client Connection Event Handler
2//!
3//! This module provides a network kernel that executes custom logic whenever a client
4//! establishes a connection. It's particularly useful for implementing server-side
5//! connection handling, authentication, and session initialization.
6//!
7//! # Features
8//! - Custom connection handling
9//! - Asynchronous event processing
10//! - Type-safe callback execution
11//! - Session security management
12//! - UDP channel support
13//! - Service discovery integration
14//!
15//! # Example:
16//! ```rust
17//! use citadel_sdk::prelude::*;
18//! use citadel_sdk::prefabs::server::client_connect_listener::ClientConnectListenerKernel;
19//!
20//! # fn main() -> Result<(), NetworkError> {
21//! let kernel = Box::new(ClientConnectListenerKernel::<_, _, StackedRatchet>::new(|conn| async move {
22//!     println!("Client connected!");
23//!     Ok(())
24//! }));
25//! # Ok(())
26//! # }
27//! ```
28//!
29//! # Important Notes
30//! - Callbacks must be Send + Sync
31//! - Futures must be Send + Sync
32//! - Handles both TCP and UDP channels
33//! - Automatic security settings handling
34//!
35//! # Related Components
36//! - [`NetKernel`]: Base trait for network kernels
37//! - [`ClientServerRemote`]: Client-server communication
38//! - [`CitadelClientServerConnection`]: Connection event data
39//! - [`NodeResult`]: Network event handling
40//!
41//! [`NetKernel`]: crate::prelude::NetKernel
42//! [`ClientServerRemote`]: crate::prelude::ClientServerRemote
43//! [`CitadelClientServerConnection`]: crate::prelude::CitadelClientServerConnection
44//! [`NodeResult`]: crate::prelude::NodeResult

46use crate::prefabs::ClientServerRemote;
47use crate::prelude::*;
48use citadel_proto::prelude::async_trait;
49use futures::Future;
50use std::marker::PhantomData;
51
52/// A kernel that executes a user-provided function each time
53/// a client makes a connection
54pub struct ClientConnectListenerKernel<F, Fut, R: Ratchet> {
55    on_channel_received: F,
56    node_remote: Option<NodeRemote<R>>,
57    _pd: PhantomData<Fut>,
58}
59
60impl<F, Fut, R: Ratchet> ClientConnectListenerKernel<F, Fut, R>
61where
62    F: Fn(CitadelClientServerConnection<R>) -> Fut + Send + Sync,
63    Fut: Future<Output = Result<(), NetworkError>> + Send + Sync,
64{
65    pub fn new(on_channel_received: F) -> Self {
66        Self {
67            on_channel_received,
68            node_remote: None,
69            _pd: Default::default(),
70        }
71    }
72}
73
74#[async_trait]
75impl<F, Fut, R: Ratchet> NetKernel<R> for ClientConnectListenerKernel<F, Fut, R>
76where
77    F: Fn(CitadelClientServerConnection<R>) -> Fut + Send + Sync,
78    Fut: Future<Output = Result<(), NetworkError>> + Send + Sync,
79{
80    fn load_remote(&mut self, server_remote: NodeRemote<R>) -> Result<(), NetworkError> {
81        self.node_remote = Some(server_remote);
82        Ok(())
83    }
84
85    async fn on_start(&self) -> Result<(), NetworkError> {
86        Ok(())
87    }
88
89    async fn on_node_event_received(&self, message: NodeResult<R>) -> Result<(), NetworkError> {
90        match message {
91            NodeResult::ConnectSuccess(ConnectSuccess {
92                ticket: _,
93                session_cid: cid,
94                remote_addr: _,
95                is_personal: _,
96                v_conn_type: conn_type,
97                services,
98                welcome_message: _,
99                channel,
100                udp_rx_opt: udp_channel_rx,
101                session_security_settings,
102            }) => {
103                let client_server_remote = ClientServerRemote::new(
104                    conn_type,
105                    self.node_remote.clone().unwrap(),
106                    session_security_settings,
107                    None,
108                    None,
109                );
110                (self.on_channel_received)(CitadelClientServerConnection {
111                    remote: client_server_remote.clone(),
112                    channel: Some(*channel),
113                    udp_channel_rx,
114                    services,
115                    cid,
116                    session_security_settings,
117                })
118                .await
119            }
120
121            other => {
122                log::trace!(target: "citadel", "Unhandled server signal: {other:?}");
123                Ok(())
124            }
125        }
126    }
127
128    async fn on_stop(&mut self) -> Result<(), NetworkError> {
129        Ok(())
130    }
131}
