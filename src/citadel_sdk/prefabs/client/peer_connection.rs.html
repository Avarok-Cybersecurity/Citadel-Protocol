1//! Peer-to-Peer Connection Management
2//!
3//! This module provides functionality for establishing and managing peer-to-peer connections
4//! in the Citadel Protocol. It supports both direct and NAT-traversed connections with
5//! configurable security settings and file transfer capabilities.
6//!
7//! # Features
8//! - Multiple simultaneous peer connections
9//! - Configurable UDP and security settings per peer
10//! - Built-in file transfer support
11//! - Automatic peer registration handling
12//! - Session password protection
13//! - Connection state management
14//! - Flexible peer identification
15//!
16//! # Example
17//! ```rust
18//! use citadel_sdk::prelude::*;
19//! use citadel_sdk::prefabs::client::peer_connection::{PeerConnectionKernel, PeerConnectionSetupAggregator};
20//!
21//! # fn main() -> Result<(), NetworkError> {
22//! async fn connect_to_peers() -> Result<(), NetworkError> {
23//!     // Set up connections to multiple peers with different settings
24//!     let peers = PeerConnectionSetupAggregator::default()
25//!         .with_peer_custom("alice")
26//!         .with_udp_mode(UdpMode::Enabled)
27//!         .add()
28//!         .with_peer_custom("bob")
29//!         .with_session_security_settings(Default::default())
30//!         .add();
31//!
32//!     let settings = DefaultServerConnectionSettingsBuilder::transient("127.0.0.1:25021")
33//!         .build()?;
34//!
35//!     let kernel = PeerConnectionKernel::new(
36//!         settings,
37//!         peers,
38//!         |connections, _remote| async move {
39//!             println!("Attemping to connect to {} peers!", connections.len());
40//!             Ok(())
41//!         },
42//!     );
43//!
44//!     Ok(())
45//! }
46//! # Ok(())
47//! # }
48//! ```
49//!
50//! # Important Notes
51//! - Peers must be mutually registered before connecting
52//! - UDP mode affects NAT traversal capabilities
53//! - File transfers require proper handler setup
54//! - Session passwords must match on both peers
55//!
56//! # Related Components
57//! - [`PeerConnectionSetupAggregator`]: Peer connection configuration
58//! - [`FileTransferHandleRx`]: File transfer handling
59//! - [`UserIdentifier`]: Peer identification
60//! - [`SessionSecuritySettings`]: Connection security
61//!
62//! [`PeerConnectionSetupAggregator`]: crate::prefabs::client::peer_connection::PeerConnectionSetupAggregator
63//! [`FileTransferHandleRx`]: crate::prefabs::client::peer_connection::FileTransferHandleRx
64//! [`UserIdentifier`]: crate::prelude::UserIdentifier
65//! [`SessionSecuritySettings`]: crate::prelude::SessionSecuritySettings

82/// After establishing a connection to the central node, this kernel begins connecting to the desired
83/// peer(s)
84pub struct PeerConnectionKernel<'a, F, Fut, R: Ratchet> {

109impl FileTransferHandleRx {
110    /// Accepts all incoming file transfer handles and processes them in the background
111    pub fn accept_all(mut self) {
112        let task = tokio::task::spawn(async move {
113            let rx = &mut self.inner;
114            while let Some(mut handle) = rx.recv().await {
115                let task = tokio::task::spawn(async move {
116                    if let Err(err) = handle.exhaust_stream().await {
117                        let orientation = handle.orientation;
118                        log::warn!(target: "citadel", "Error background handling of file transfer for {orientation:?}: {err:?}");
119                    }
120                });
121
122                drop(task);
123            }
124        });
125
126        drop(task);
127    }
128}

150#[async_trait]
151impl<F, Fut, R: Ratchet> NetKernel<R> for PeerConnectionKernel<'_, F, Fut, R> {
152    fn load_remote(&mut self, server_remote: NodeRemote<R>) -> Result<(), NetworkError> {
153        self.inner_kernel.load_remote(server_remote)
154    }
155
156    async fn on_start(&self) -> Result<(), NetworkError> {
157        self.inner_kernel.on_start().await
158    }
159
160    #[allow(clippy::collapsible_else_if)]
161    async fn on_node_event_received(&self, message: NodeResult<R>) -> Result<(), NetworkError> {
162        match message {
163            NodeResult::ObjectTransferHandle(ObjectTransferHandle {
164                ticket: _,
165                handle,
166                session_cid,
167            }) => {
168                let is_revfs = matches!(
169                    handle.metadata.transfer_type,
170                    TransferType::RemoteEncryptedVirtualFilesystem { .. }
171                );
172                let active_peers = self.shared.active_peer_conns.lock();
173                let v_conn = if is_revfs {
174                    let peer_cid = if session_cid != handle.source {
175                        handle.source
176                    } else {
177                        handle.receiver
178                    };
179                    PeerConnectionType::LocalGroupPeer {
180                        session_cid,
181                        peer_cid,
182                    }
183                } else {
184                    if matches!(
185                        handle.orientation,
186                        ObjectTransferOrientation::Receiver { .. }
187                    ) {
188                        PeerConnectionType::LocalGroupPeer {
189                            session_cid,
190                            peer_cid: handle.source,
191                        }
192                    } else {
193                        PeerConnectionType::LocalGroupPeer {
194                            session_cid,
195                            peer_cid: handle.receiver,
196                        }
197                    }
198                };
199
200                if let Some(peer_ctx) = active_peers.get(&v_conn) {
201                    if let Err(err) = peer_ctx.send_file_transfer_tx.send(handle) {
202                        log::warn!(target: "citadel", "Error forwarding file transfer handle: {:?}", err.to_string());
203                    }
204                } else {
205                    log::warn!(target: "citadel", "Unable to find key for inbound file transfer handle: {:?}\n Active Peers: {:?} \n handle_source = {}, handle_receiver = {}", v_conn, active_peers.keys().cloned().collect::<Vec<_>>(), handle.source, handle.receiver);
206                }
207
208                Ok(())
209            }
210
211            // Note: NodeResult::Disconnect is for C2S connections only.
212            // P2P disconnects are handled via NodeResult::PeerEvent(PeerSignal::Disconnect).
213            // C2S disconnects don't require removing from active_peer_conns.
214            unprocessed @ NodeResult::Disconnect(..) | unprocessed => {
215                // pass any unprocessed events to the lower kernel
216                self.inner_kernel.on_node_event_received(unprocessed).await
217            }
218        }
219    }
220
221    async fn on_stop(&mut self) -> Result<(), NetworkError> {
222        self.inner_kernel.on_stop().await
223    }
224}

242pub struct AddedPeer {
243    list: PeerConnectionSetupAggregator,
244    id: UserIdentifier,
245    session_security_settings: Option<SessionSecuritySettings>,
246    ensure_registered: bool,
247    udp_mode: Option<UdpMode>,
248    peer_session_password: Option<PreSharedKey>,

251impl AddedPeer {
252    /// Adds the peer
253    pub fn add(mut self) -> PeerConnectionSetupAggregator {
254        let new = PeerConnectionSettings {
255            id: self.id,
256            session_security_settings: self.session_security_settings.unwrap_or_default(),
257            udp_mode: self.udp_mode.unwrap_or_default(),
258            ensure_registered: self.ensure_registered,
259            peer_session_password: self.peer_session_password,
260        };
261
262        self.list.inner.push(new);
263        self.list
264    }
265
266    /// Sets the [`UdpMode`] for this peer to peer connection
267    pub fn with_udp_mode(mut self, udp_mode: UdpMode) -> Self {
268        self.udp_mode = Some(udp_mode);
269        self
270    }
271
272    /// Disables the UDP mode for the client-to-server connection. The default setting is Disabled
273    pub fn disable_udp(self) -> Self {
274        self.with_udp_mode(UdpMode::Disabled)
275    }
276
277    /// Enables the UDP mode for the client-to-server connection. The default setting is Disabled
278    pub fn enable_udp(self) -> Self {
279        self.with_udp_mode(UdpMode::Enabled)
280    }
281
282    /// Sets the [`SessionSecuritySettings`] for this peer to peer connection
283    pub fn with_session_security_settings(
284        mut self,
285        session_security_settings: SessionSecuritySettings,
286    ) -> Self {
287        self.session_security_settings = Some(session_security_settings);
288        self
289    }
290
291    /// Ensures that the target user is registered before attempting to connect
292    pub fn ensure_registered(mut self) -> Self {
293        self.ensure_registered = true;
294        self
295    }
296
297    /// Adds a pre-shared key to the peer session password list. Both connecting nodes
298    /// must have matching passwords in order to establish a connection. Default is None.
299    pub fn with_session_password<T: Into<PreSharedKey>>(mut self, password: T) -> Self {
300        self.peer_session_password = Some(password.into());
301        self
302    }
303}

305impl PeerConnectionSetupAggregator {
306    /// Adds a peer with default connection settings
307    /// ```
308    /// use citadel_sdk::prelude::*;
309    /// let peers = PeerConnectionSetupAggregator::default()
310    ///     .with_peer("john.doe")
311    ///     .with_peer("alice")
312    ///     .with_peer("bob");
313    /// ```
314    pub fn with_peer<T: Into<UserIdentifier>>(self, peer: T) -> PeerConnectionSetupAggregator {
315        self.with_peer_custom(peer).add()
316    }
317
318    /// Adds a peer with custom settings
319    /// ```
320    /// use citadel_sdk::prelude::*;
321    /// // Set up a p2p connection to john.doe with udp enabled,
322    /// // and, a p2p connection to alice with udp disabled and
323    /// // custom security settings
324    /// let peers = PeerConnectionSetupAggregator::default()
325    ///     .with_peer_custom("john.doe")
326    ///     .with_udp_mode(UdpMode::Enabled)
327    ///     .add()
328    ///     .with_peer_custom("alice")
329    ///     .with_udp_mode(UdpMode::Disabled)
330    ///     .with_session_security_settings(Default::default())
331    ///     .add();
332    /// ```
333    pub fn with_peer_custom<T: Into<UserIdentifier>>(self, peer: T) -> AddedPeer {
334        AddedPeer {
335            list: self,
336            id: peer.into(),
337            ensure_registered: false,
338            session_security_settings: None,
339            udp_mode: None,
340            peer_session_password: None,
341        }
342    }

396#[async_trait]
397impl<'a, F, Fut, T: Into<PeerConnectionSetupAggregator> + Send + 'a, R: Ratchet>
398    PrefabFunctions<'a, T, R> for PeerConnectionKernel<'a, F, Fut, R>
399where
400    F: FnOnce(
401            Receiver<Result<PeerConnectSuccess<R>, NetworkError>>,
402            CitadelClientServerConnection<R>,
403        ) -> Fut
404        + Send
405        + 'a,
406    Fut: Future<Output = Result<(), NetworkError>> + Send + 'a,
407{
408    type UserLevelInputFunction = F;
409    type SharedBundle = Shared;
410
411    fn get_shared_bundle(&self) -> Self::SharedBundle {
412        self.shared.clone()
413    }
414
415    #[allow(clippy::blocks_in_conditions)]
416    #[cfg_attr(
417        feature = "localhost-testing",
418        tracing::instrument(level = "trace", target = "citadel", skip_all, ret, err(Debug))
419    )]
420    async fn on_c2s_channel_received(
421        connect_success: CitadelClientServerConnection<R>,
422        peers_to_connect: T,
423        f: Self::UserLevelInputFunction,
424        shared: Shared,
425    ) -> Result<(), NetworkError> {
426        let shared = &shared;
427        let session_cid = connect_success.cid;
428        let mut peers_already_registered = vec![];
429
430        wait_for_peers().await;
431        let peers_to_connect = peers_to_connect.into().inner;
432
433        for peer in &peers_to_connect {
434            // TODO: optimize this into a single concurrent operation
435            peers_already_registered.push(
436                peer.id
437                    .search_peer(session_cid, connect_success.account_manager())
438                    .await?,
439            )
440        }
441
442        let remote = connect_success.clone();
443        let (tx, rx) = citadel_io::tokio::sync::mpsc::channel(peers_to_connect.len());
444        let requests = FuturesUnordered::new();
445
446        for (mutually_registered, peer_to_connect) in
447            peers_already_registered.into_iter().zip(peers_to_connect)
448        {
449            // Each task will be responsible for possibly registering to and connecting
450            // with the desired peer
451            let remote = remote.clone();
452            let tx = tx.clone();
453            let PeerConnectionSettings {
454                id,
455                session_security_settings,
456                udp_mode,
457                ensure_registered,
458                peer_session_password,
459            } = peer_to_connect;
460
461            let task = async move {
462                let inner_task = async move {
463                    let (file_transfer_tx, file_transfer_rx) =
464                        citadel_io::tokio::sync::mpsc::unbounded_channel();
465
466                    // Get the actual peer CID from the mutual registration info if available
467                    let peer_cid = if let Some(mutual_peer) = &mutually_registered {
468                        mutual_peer.cid
469                    } else {
470                        id.get_cid()
471                    };
472
473                    let handle = if let Some(_already_registered) = mutually_registered {
474                        remote.find_target(session_cid, id).await?
475                    } else {
476                        // TODO: optimize peer registration + connection in one go
477                        log::info!(target: "citadel", "{session_cid} proposing target {id:?} to central node");
478                        let handle = remote.propose_target(session_cid, id.clone()).await?;
479                        // if the peer is not yet registered to the central node, wait for it to become registered
480                        // this is useful especially for testing purposes
481                        if ensure_registered {
482                            loop {
483                                if handle.is_peer_registered().await? {
484                                    break;
485                                }
486                                citadel_io::tokio::time::sleep(std::time::Duration::from_millis(
487                                    200,
488                                ))
489                                .await;
490                            }
491                        }
492
493                        log::info!(target: "citadel", "{session_cid} registering to peer {id:?}");
494                        let _reg_success = handle.register_to_peer().await?;
495                        log::info!(target: "citadel", "{session_cid} registered to peer {id:?} registered || success -> now connecting");
496                        handle
497                    };
498
499                    // Register the peer connection early before attempting to connect
500                    // This prevents race conditions where file transfers arrive before connection completes
501                    let peer_conn = PeerConnectionType::LocalGroupPeer {
502                        session_cid,
503                        peer_cid,
504                    };
505                    let peer_context = PeerContext {
506                        conn_type: peer_conn,
507                        send_file_transfer_tx: file_transfer_tx.clone(),
508                    };
509                    log::debug!(target: "citadel", "Early registering peer connection: {peer_conn:?}");
510                    let _ = shared
511                        .active_peer_conns
512                        .lock()
513                        .insert(peer_conn, peer_context);
514
515                    handle
516                        .connect_to_peer_custom(
517                            session_security_settings,
518                            udp_mode,
519                            peer_session_password,
520                        )
521                        .await
522                        .map(|mut success| {
523                            let actual_peer_conn = success.channel.get_peer_conn_type().unwrap();
524
525                            // If the actual peer connection type differs from our early registration,
526                            // update it
527                            if actual_peer_conn != peer_conn {
528                                log::debug!(target: "citadel", "Updating peer connection registration from {peer_conn:?} to {actual_peer_conn:?}");
529                                let mut active_peers = shared.active_peer_conns.lock();
530                                if let Some(peer_ctx) = active_peers.remove(&peer_conn) {
531                                    let _ = active_peers.insert(actual_peer_conn, peer_ctx);
532                                }
533                            }
534                            // Update the existing entry with the file transfer receiver
535                            success.incoming_object_transfer_handles = Some(FileTransferHandleRx {
536                                inner: file_transfer_rx,
537                                conn_type: actual_peer_conn.as_virtual_connection(),
538                            });
539                            success
540                        })
541                        .inspect_err(|_err| {
542                            // Clean up the early registration on connection failure
543                            let _ = shared.active_peer_conns.lock().remove(&peer_conn);
544                        })
545                };
546
547                tx.send(inner_task.await)
548                    .await
549                    .map_err(|err| NetworkError::Generic(err.to_string()))
550            };
551
552            requests.push(Box::pin(task))
553        }
554
555        // TODO: What should be done if a peer conn fails? No room for error here
556        // Drop the original tx so channel closes after all tasks complete
557        drop(tx);
558
559        // Use join! not try_join! to ensure both branches complete even if one errors
560        // This prevents premature cancellation of the user callback
561        let (collection_result, user_result) =
562            citadel_io::tokio::join!(requests.try_collect::<()>(), f(rx, connect_success));
563
564        // Return first error, or success if both succeeded
565        collection_result?;
566        user_result
567    }
568
569    fn construct(kernel: Box<dyn NetKernel<R> + 'a>) -> Self {
570        Self {
571            inner_kernel: kernel,
572            shared: Shared {
573                active_peer_conns: Arc::new(Mutex::new(Default::default())),
574            },
575            _pd: Default::default(),
576        }
577    }
578}

1301        mut conn_rx: Receiver<Result<PeerConnectSuccess<R>, NetworkError>>,
1302        session_cid: u64,
1303        peer_count: usize,
1304        udp_mode: UdpMode,
1305        checks: F,
1306    ) -> Vec<PeerConnectSuccess<R>>
1307    where
1308        F: Fn(PeerConnectSuccess<R>) -> Fut + Send + Clone + 'static,
1309        Fut: Future<Output = PeerConnectSuccess<R>> + Send,
1310    {
1311        let (finished_tx, finished_rx) = tokio::sync::oneshot::channel();
1312
1313        let task = async move {
1314            let (done_tx, mut done_rx) = tokio::sync::mpsc::unbounded_channel();
1315            let mut conns = vec![];
1316            while let Some(conn) = conn_rx.recv().await {
1317                conns.push(conn);
1318                if conns.len() == peer_count - 1 {
1319                    break;
1320                }
1321            }
1322
1323            log::info!(target: "citadel", "~~~*** Peer {session_cid} has {} connections to other peers ***~~~", conns.len());
1324
1325            for conn in conns {
1326                let conn = conn.expect("Error receiving peer connection");
1327                handle_peer_connect_success(
1328                    conn,
1329                    done_tx.clone(),
1330                    session_cid,
1331                    udp_mode,
1332                    checks.clone(),
1333                );
1334            }
1335
1336            // Now, wait for all to finish
1337            let mut ret = vec![];
1338            while let Some(done) = done_rx.recv().await {
1339                ret.push(done);
1340                if ret.len() == peer_count - 1 {
1341                    break;
1342                }
1343            }
1344
1345            finished_tx
1346                .send(ret)
1347                .expect("Error sending finished signal in handle_peer_connect_successes");
1348        };
1349
1350        drop(tokio::task::spawn(task));
1351        let ret = finished_rx
1352            .await
1353            .expect("Error receiving finished signal in handle_peer_connect_successes");
1354
1355        assert_eq!(ret.len(), peer_count - 1);
1356        ret
1357    }

1359    fn handle_peer_connect_success<F, Fut, R: Ratchet>(
1360        mut conn: PeerConnectSuccess<R>,
1361        done_tx: UnboundedSender<PeerConnectSuccess<R>>,
1362        session_cid: u64,
1363        udp_mode: UdpMode,
1364        checks: F,
1365    ) where
1366        F: Fn(PeerConnectSuccess<R>) -> Fut + Send + Clone + 'static,
1367        Fut: Future<Output = PeerConnectSuccess<R>> + Send,
1368    {
1369        let task = async move {
1370            let chan = conn.udp_channel_rx.take();
1371            crate::test_common::p2p_assertions(session_cid, &conn).await;
1372            crate::test_common::udp_mode_assertions(udp_mode, chan).await;
1373            let conn = checks(conn).await;
1374            done_tx
1375                .send(conn)
1376                .expect("Error sending done signal in handle_peer_connect_success");
1377        };
1378
1379        drop(tokio::task::spawn(task));
1380    }
